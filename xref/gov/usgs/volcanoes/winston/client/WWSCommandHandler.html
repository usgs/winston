1   package gov.usgs.volcanoes.winston.client;
2   
3   import java.io.IOException;
4   import java.util.concurrent.Semaphore;
5   
6   /**
7    * Receive and process response from winston.
8    *
9    * @author Tom Parker
10   */
11  public abstract class WWSCommandHandler {
12  	/**
13  	 * Semaphore indicating if a read has been completed. While a request is
14  	 * being processed, attempts to acquire the semaphore will block.
15  	 */
16  	protected final Semaphore sem;
17  
18  	public WWSCommandHandler() {
19  		sem = new Semaphore(0);
20  	}
21  
22  	/** Process response from winston */
23  	public abstract void handle(Object msg) throws IOException;
24  
25  	/**
26  	 * Block until the handler has received and processed server response.
27  	 * 
28  	 * @throws InterruptedException
29  	 *             when receives InterruptedException
30  	 */
31  	public void responseWait() throws InterruptedException {
32  		sem.acquire();
33  	}
34  }

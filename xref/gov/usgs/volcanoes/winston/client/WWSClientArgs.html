7   package gov.usgs.volcanoes.winston.client;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import com.martiansoftware.jsap.FlaggedOption;
13  import com.martiansoftware.jsap.JSAP;
14  import com.martiansoftware.jsap.JSAPResult;
15  import com.martiansoftware.jsap.Parameter;
16  import com.martiansoftware.jsap.Switch;
17  
18  import gov.usgs.volcanoes.core.args.Args;
19  import gov.usgs.volcanoes.core.args.Arguments;
20  import gov.usgs.volcanoes.core.args.decorator.ScnlArg;
21  import gov.usgs.volcanoes.core.args.decorator.TimeSpanArg;
22  import gov.usgs.volcanoes.core.args.decorator.VerboseArg;
23  import gov.usgs.volcanoes.core.data.Scnl;
24  import gov.usgs.volcanoes.core.time.TimeSpan;
25  import gov.usgs.volcanoes.core.util.StringUtils;
26  
27  /**
28   * Argument processor for WWSClient.
29   *
30   * @author Tom Parker
31   */
32  public class WWSClientArgs {
33  	private static final Logger LOGGER = LoggerFactory.getLogger(WWSClientArgs.class);
34  
35  	/** format of time on cmd line */
36  	public static final String INPUT_TIME_FORMAT = "yyyyMMddHHmm";
37  
38  	private static final String PROGRAM_NAME = "java -jar gov.usgs.volcanoes.winston.WWSClient";
39  	private static final String EXPLANATION = "I am the Winston Wave Server client.\n";
40  
41  	private static final Parameter[] PARAMETERS = new Parameter[] {
42  			new FlaggedOption("server", JSAP.STRING_PARSER, JSAP.NO_DEFAULT, JSAP.REQUIRED, 's', "server", "Remote server."),
43  			new FlaggedOption("port", JSAP.INTEGER_PARSER, "16022", JSAP.NOT_REQUIRED, 'p', "port", "Remote port."),
44  			new Switch("sac", JSAP.NO_SHORTFLAG, "sac", "Write SAC output."),
45  			new Switch("txt", JSAP.NO_SHORTFLAG, "txt", "Write samples as text output."),
46  			new Switch("rsam", JSAP.NO_SHORTFLAG, "rsam", "Write RSAM output."),
47  			new FlaggedOption("rsamPeriod", JSAP.INTEGER_PARSER, "300", JSAP.NOT_REQUIRED, JSAP.NO_SHORTFLAG, "rsamPeriod", "RSAM period."),
48  			new Switch("heli", JSAP.NO_SHORTFLAG, "heli", "Write Helicorder output."),
49  			new Switch("menu", JSAP.NO_SHORTFLAG, "menu", "Retrieve server menu.") };
50  
51  	/** If true, log more. */
52  	public final boolean verbose;
53  
54  	/** Remote server. */
55  	public final String server;
56  
57  	/** Remote server port. */
58  	public final int port;
59  
60  	/** Time span to retrieve. */
61  	public final TimeSpan timeSpan;
62  
63  	/** channel to query */
64  	public final Scnl channel;
65  
66  	/** If true, print server menu */
67  	public final boolean menu;
68  	
69  	/** If true, write SAC output. */
70  	public final boolean sacOutput;
71  
72  	/** If true, write TXT output. */
73  	public final boolean txtOutput;
74  
75  	/** If true, write RSAM output. */
76  	public final boolean rsamOutput;
77  
78  	/** RSAM period in seconds*/
79  	public final int rsamPeriod;
80  	
81  	/** If true, write Heli output. */
82  	public final boolean heliOutput;
83  
84  	/**
85  	 * Class constructor.
86  	 * 
87  	 * @param commandLineArgs
88  	 *            the command line arguments
89  	 * @throws Exception
90  	 *             when things go wrong
91  	 */
92  	public WWSClientArgs(final String[] commandLineArgs) throws Exception {
93  		Arguments args = null;
94  		args = new Args(PROGRAM_NAME, EXPLANATION, PARAMETERS);
95  		args = new ScnlArg(false, args);
96  		args = new TimeSpanArg(INPUT_TIME_FORMAT, false, args);
97  		args = new VerboseArg(args);
98  
99  		JSAPResult jsapResult = null;
100 		jsapResult = args.parse(commandLineArgs);
101 
102 		verbose = jsapResult.getBoolean("verbose");
103 		timeSpan = (TimeSpan) jsapResult.getObject("timeSpan");
104 		server = jsapResult.getString("server");
105 		port = jsapResult.getInt("port");
106 		channel = (Scnl) jsapResult.getObject("channel");
107 		menu = jsapResult.getBoolean("menu");
108 		sacOutput = jsapResult.getBoolean("sac");
109 		txtOutput = jsapResult.getBoolean("txt");
110 		rsamOutput = jsapResult.getBoolean("rsam");
111 		rsamPeriod = jsapResult.getInt("rsamPeriod");
112 		heliOutput = jsapResult.getBoolean("heli");
113 		
114 		if (!jsapResult.getBoolean("help")) {
115 			LOGGER.debug("Setting: verbose={}", verbose);
116 			LOGGER.debug("Setting: timeSpan={}", timeSpan);
117 			LOGGER.debug("Setting: server={}", server);
118 			LOGGER.debug("Setting: port={}", port);
119 			LOGGER.debug("Setting menu={}", menu);
120 			LOGGER.debug("Setting sacOutput={}", sacOutput);
121 			LOGGER.debug("Setting txtOutput={}", txtOutput);
122 			LOGGER.debug("Setting rsamOutput={}", rsamOutput);
123 			LOGGER.debug("Setting heliOutput={}", heliOutput);
124 		}
125 
126 		if (sacOutput || txtOutput || rsamOutput || heliOutput) {
127 			if (channel == null) {
128 				throw new RuntimeException("No channel provided.");
129 			}
130 			if (timeSpan == null) {
131 				throw new RuntimeException("No time span provided.");
132 			}
133 		}
134 		
135 	}
136 }

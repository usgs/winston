37    public class EwBox extends JTextField implements Observer {
38  
39      private static final long serialVersionUID = 1L;
40  
41      public EwBox() {
42        super(20);
43        time.addObserver(this);
44        setToolTipText("seconds past 1970-01-01 00:00:00 UTC");
45  
46  
47        addActionListener(new ActionListener() {
48          public void actionPerformed(final ActionEvent evt) {
49            updateTime(getText());
50          }
51        });
52        addFocusListener(new FocusListener() {
53          public void focusGained(final FocusEvent e) {}
54  
55          public void focusLost(final FocusEvent e) {
56            updateTime(getText());
57          }
58        });
59  
60      }
61  
62      public void update(final Observable o, final Object arg) {
63        final double d = ((TimeSubject) o).getEW();
64        setText(timeFormatter.format(d));
65        setBackground(WHITE);
66      }
67  
68      public final void updateTime(final String s) {
69        try {
70          time.setDate(Ew.asDate(Double.parseDouble(s)));
71        } catch (final NumberFormatException ex) {
72          setBackground(RED);
73        }
74      }
75    }

94            updateTime(getText());

100       setText("" + ((TimeSubject) o).getInputTime());

104     public final void updateTime(final String s) {
105       SimpleDateFormat dateF = new SimpleDateFormat(Time.INPUT_TIME_FORMAT);
106       Date d;
107       try {
108         d = dateF.parse(getText());
109         time.setDate(d);
110       } catch (ParseException e) {
111         setBackground(RED);
112       }
113     }

194   public class TimeSubject extends Observable {
195 
196     Date date;
197 
198     public TimeSubject() {
199       setDate();
200     }
201 
202 
203     public double getEW() {
204       return gov.usgs.volcanoes.core.time.Ew.fromEpoch(date.getTime());
205     }
206 
207     public String getInputTime() {
208       return Time.format(Time.INPUT_TIME_FORMAT, date);
209     }
210 
211     public double getJ2k() {
212       return J2kSec.fromDate(date);
213     }
214 
215     public String getStandardTime() {
216       return Time.format(Time.STANDARD_TIME_FORMAT, date);
217     }
218 
219     public void setDate() {
220       setDate(new Date());
221     }
222 
223     public void setDate(final Date d) {
224       date = d;
225       setChanged();
226       notifyObservers();
227     }
228   }

270   @Override

275     final FormLayout layout = new FormLayout("right:max(40dlu;p), 4dlu, left:p", "");
276 
277     final DefaultFormBuilder builder = new DefaultFormBuilder(layout);
278     builder.setDefaultDialogBorder();
279     builder.appendSeparator("Time Formats");
280     builder.append("Standard", standardF);
281     builder.nextLine();
282     builder.append("Input", inputF);
283     builder.nextLine();
284     builder.append("J2K", j2kF);
285     builder.nextLine();
286     builder.append("EW", ewF);
287     builder.nextLine();
288     builder.appendUnrelatedComponentsGapRow();
289     builder.nextLine();
290     builder.append("", nowB);

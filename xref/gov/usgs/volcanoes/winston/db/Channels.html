45    /**
46     * Get map of group nodes
47     * 
48     * @return Map of ids to group nodes
49     */
50    public Map<Integer, GroupNode> getGroupNodes() {
51      if (!winston.checkConnect())
52        return null;
53  
54      try {
55        final HashMap<Integer, GroupNode> result = new HashMap<Integer, GroupNode>();
56        winston.useRootDatabase();
57        final ResultSet rs = winston.executeQuery("SELECT * FROM groupnodes");
58        while (rs.next()) {
59          final GroupNode gn = new GroupNode(rs);
60          result.put(gn.nid, gn);
61        }
62        GroupNode.buildTree(result);
63        rs.close();
64  
65        return result;
66      } catch (final Exception e) {
67        LOGGER.error("Could not get groups.");
68      }
69      return null;
70    }

85    /**
86     * Get List of channels sorted by data age
87     * 
88     * @return List of channels
89     */
90    public List<Channel> getChannelsByLastInsert() {
91      final List<Channel> sts = getChannels();
92  
93      Collections.sort(sts, new Comparator<Channel>() {
94        public int compare(final Channel c1, final Channel c2) {
95          final Double t1 = c1.getMaxTime();
96          final Double t2 = c2.getMaxTime();
97          return t2.compareTo(t1);
98        }
99      });
100 
101     return sts;
102   }

104   /**
105    * Get List of channels
106    * 
107    * @param fullMetadata
108    *          if true, included full metadata
109    * @return List of channels
110    */
111   public List<Channel> getChannels(final boolean fullMetadata) {
112     if (!winston.checkConnect())
113       return null;
114 
115     try {
116       winston.useRootDatabase();
117       ResultSet rs = winston.executeQuery(
118           "SELECT sid, instruments.iid, code, alias, unit, linearA, linearB, st, et, instruments.lon, instruments.lat, height, name, description, timezone "
119               + "FROM channels LEFT JOIN instruments ON channels.iid=instruments.iid "
120               + "ORDER BY code ASC");
121       final Map<Integer, Channel> channelsMap = new HashMap<Integer, Channel>();
122       final List<Channel> channels = new ArrayList<Channel>();
123       while (rs.next()) {
124         final Channel ch = new Channel(rs, aparentRetention);
125         
126         if (ch.getMaxTime() > ch.getMinTime()) {
127           channelsMap.put(ch.getSID(), ch);
128           channels.add(ch);
129         }
130       }
131       rs.close();
132       final Map<Integer, GroupNode> nodes = getGroupNodes();
133 
134       rs = winston.executeQuery("SELECT sid, nid FROM grouplinks");
135       while (rs.next()) {
136         final int sid = rs.getInt(1);
137         final int nid = rs.getInt(2);
138         final Channel ch = channelsMap.get(sid);
139         if (ch != null) {
140           final GroupNode gn = nodes.get(nid);
141           if (gn != null)
142             ch.addGroup(gn.toString());
143         }
144       }
145       rs.close();
146 
147       if (fullMetadata) {
148         final PreparedStatement ps = winston
149             .getPreparedStatement("SELECT * FROM channelmetadata WHERE sid=? ORDER BY name ASC");
150         for (final Channel ch : channels) {
151           HashMap<String, String> md = null;
152           ps.setInt(1, ch.getSID());
153           rs = ps.executeQuery();
154           while (rs.next()) {
155             if (md == null)
156               md = new HashMap<String, String>();
157             md.put(rs.getString("name"), rs.getString("value"));
158           }
159           ch.setMetadata(md);
160         }
161       }
162 
163       return channels;
164     } catch (final Exception e) {
165       LOGGER.error("Could not get channels.", e);
166     }
167     return null;
168   }

188       return result;

305   /**
306    * Create channel w/ code
307    * 
308    * @param code
309    * @return true if successful
310    */
311   public boolean createChannel(final String code) {
312     if (!winston.checkConnect())
313       return false;
314     try {
315       winston.useRootDatabase();
316       winston.getStatement()
317       .execute("CREATE DATABASE `" + winston.databasePrefix + "_" + code + "`");
318       winston.getStatement()
319           .execute("INSERT INTO channels (code, st, et) VALUES ('" + code + "', 1E300, -1E300)");
320       winston.getStatement().execute("USE `" + winston.databasePrefix + "_" + code + "`");
321     } catch (final Exception e) {
322       LOGGER.error("Could not create channel.  Are permissions set properly?");
323     }
324     return false;
325   }

327   /**
328    * Create or update row in instrument table
329    * 
330    * TODO: add network code once winston schema supports it
331    * 
332    * @param inst
333    * @return true if successful
334    */
335   public void updateInstrument(final Instrument inst) {
336     if (!winston.checkConnect())
337       return;
338     try {
339       winston.useRootDatabase();
340       int iid = getInstrumentId(inst.getName());
341       PreparedStatement ps = null;
342       final boolean instrumentExists = iid > 0;
343       if (instrumentExists) {
344         ps = winston.getPreparedStatement(
345             "UPDATE instruments SET name=?, description=?, lon=?, lat=?, height=? where iid=?;");
346         ps.setInt(6, iid);
347       } else {
348         ps = winston.getPreparedStatement(
349             "INSERT INTO instruments (name, description, lon, lat, height) VALUES (?,?,?,?,?);");
350       }
351       ps.setString(1, inst.getName());
352       ps.setString(2, inst.getDescription());
353       ps.setDouble(3, inst.getLongitude());
354       ps.setDouble(4, inst.getLatitude());
355       ps.setDouble(5, inst.getHeight());
356       LOGGER.debug(ps.toString());
357       ps.execute();
358 
359       if (!instrumentExists) {
360         iid = getInstrumentId(inst.getName());
361         ps = winston.getPreparedStatement("UPDATE channels set iid=? WHERE code LIKE ?;");
362         ps.setInt(1, iid);
363         ps.setString(2, inst.getName() + "$%");
364         LOGGER.debug(ps.toString());
365 
366         ps.execute();
367       }
368     } catch (final Exception e) {
369       LOGGER.error("Could not create channel.  Are permissions set properly? ({})", e.getMessage());
370       System.exit(1);
371     }
372   }

374   /**
375    * Getter for instruments
376    * 
377    * @return List of instruments
378    */
379   public List<Instrument> getInstruments() {
380     if (!winston.checkConnect())
381       return null;
382 
383     try {
384       winston.useRootDatabase();
385       ResultSet rs = winston.executeQuery("SELECT * FROM instruments ORDER BY name ASC");
386       final List<Instrument> insts = new ArrayList<Instrument>();
387       while (rs.next()) {
388         final Instrument inst = new Instrument(rs);
389         insts.add(inst);
390       }
391       rs.close();
392 
393       final PreparedStatement ps = winston
394           .getPreparedStatement("SELECT * FROM instrumentmetadata WHERE iid=? ORDER BY name ASC");
395       for (final Instrument inst : insts) {
396         HashMap<String, String> md = null;
397         ps.setInt(1, inst.getID());
398         rs = ps.executeQuery();
399         while (rs.next()) {
400           if (md == null)
401             md = new HashMap<String, String>();
402           md.put(rs.getString("name"), rs.getString("value"));
403         }
404         inst.setMetadata(md);
405       }
406       return insts;
407     } catch (final Exception e) {
408       LOGGER.error("Could not get instruments.");
409     }
410     return null;
411   }

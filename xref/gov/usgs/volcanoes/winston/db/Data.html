54    /**
55     * Get timespan for specific channel
56     *
57     * @param code
58     *          channel code
59     * @return Array of start and end times
60     */
61    public double[] getTimeSpan(final String code) {
62      if (!winston.checkConnect())
63        return null;
64      try {
65        final ResultSet rs = winston.getStatement().executeQuery("SELECT st, et FROM `"
66            + winston.databasePrefix + "_ROOT`.channels WHERE code='" + code + "'");
67        rs.next();
68        final double[] d = new double[] {rs.getDouble(1), rs.getDouble(2)};
69        rs.close();
70        return d;
71      } catch (final Exception e) {
72        LOGGER.error("Could not get time span for channel: {}. ({})", code, e.getLocalizedMessage());
73      }
74      return null;
75    }

89    /**
90     * Get days between times t1 & t2
91     *
92     * @param t1
93     *          start of timespan
94     * @param t2
95     *          end of timespan
96     * @return List of strings representing each day between t1 & t2
97     */
98    private List<String> daysBetween(final double t1, final double t2) {
99      final ArrayList<String> result = new ArrayList<String>();
100     double ct = t1;
101     while (ct < t2 + ONE_DAY) {
102       result.add(J2kSec.format(WinstonDatabase.WINSTON_TABLE_DATE_FORMAT, ct));
103       ct += ONE_DAY;
104     }
105     return result;
106   }

108   /**
109    * Finds data gaps in a given channel between two times. Returns null
110    * on a Winston error. Returns a single item list with the given time span
111    * if the channel doesn't exist or if no data exist in the interval.
112    *
113    * @param code
114    * @param t1
115    * @param t2
116    * @return List of data gaps (each gap by a start and end time)
117    */
118   public List<double[]> findGaps(final String code, final double t1, final double t2) {
119     if (!winston.checkConnect())
120       return null;
121 
122     final List<double[]> gaps = new ArrayList<double[]>();
123     if (!winston.useDatabase(code)) {
124       // database didn't exist so the whole thing must be a gap
125       gaps.add(new double[] {t1, t2});
126       return gaps;
127     }
128 
129     try {
130       final SimpleDateFormat dateFormat = new SimpleDateFormat("yyyy_MM_dd");
131 
132       final List<String> days = daysBetween(t1, t2);
133 
134       dateFormat.setTimeZone(TimeZone.getTimeZone("GMT"));
135       final List<double[]> bufs = new ArrayList<double[]>(2 * ONE_DAY);
136       for (final String day : days) {
137         final double tst = J2kSec.parse(WinstonDatabase.WINSTON_TABLE_DATE_FORMAT, day);
138         final double tet = tst + ONE_DAY;
139         final String table = code + "$$" + day;
140         if (!winston.tableExists(code, table))
141           continue;
142 
143         if (tet < t1)
144           continue;
145         if (tst > t2)
146           continue;
147 
148         final ResultSet rs = winston.getStatement()
149             .executeQuery("SELECT st, et FROM `" + table + "` ORDER BY st ASC");
150         while (rs.next()) {
151           final double start = rs.getDouble(1);
152           final double end = rs.getDouble(2);
153           if (end < t1)
154             continue;
155           if (start > t2)
156             continue;
157           bufs.add(new double[] {start, end});
158         }
159         rs.close();
160       }
161 
162       if (bufs == null || bufs.size() == 0) {
163         // there were no tracebufs in the time range, the whole span is a gap.
164         gaps.add(new double[] {t1, t2});
165         return gaps;
166       }
167 
168       final double epsilon = 0.01;
169       if (bufs.get(0)[0] > t1) {
170         gaps.add(new double[] {t1, bufs.get(0)[0]});
171       }
172       double last = bufs.get(0)[1];
173       for (int i = 1; i < bufs.size(); i++) {
174         final double[] buf = bufs.get(i);
175         if (buf[0] - last > epsilon) {
176           double start = last;
177           double end = buf[0];
178           if (end < t1)
179             continue;
180           if (start > t2)
181             continue;
182           if (start < t1)
183             start = t1;
184           if (end > t2)
185             end = t2;
186 
187           gaps.add(new double[] {start, end});
188         }
189 
190         last = buf[1];
191       }
192       if (bufs.get(bufs.size() - 1)[1] < t2) {
193         gaps.add(new double[] {bufs.get(bufs.size() - 1)[1], t2});
194       }
195 
196       return gaps;
197     } catch (final Exception e) {
198       e.printStackTrace();
199     }
200     return null;
201   }

224   /**
225    * Yield tracebuf from a ResultSet as an array of buytes
226    */
227   private byte[] getTraceBufBytes(final ResultSet rs) throws SQLException, IOException {
228     final Blob b = rs.getBlob("tracebuf");
229     byte[] bs = b.getBytes(1, (int) b.length());
230     bs = Zip.decompress(bs, 512);
231     return bs;
232   }

234   /**
235    * Modified 2006-08-14 to not check for consistent sample rates. That duty
236    * is passed to TraceBuf.traceBufsToWave(). This should fix the HVO bug
237    * that trig2disk can not get tracebufs with varying sample rates.
238    *
239    * @param code
240    * @param t1
241    * @param t2
242    * @param maxrows
243    * @return trace buf data
244    * @throws UtilException
245    */
246   public List<byte[]> getTraceBufBytes(final String code, final double t1, final double t2,
247       final int maxrows) throws UtilException {
248     int numSamplesCounter = 0;
249     if (!winston.checkConnect() || !winston.useDatabase(code))
250       return null;
251     try {
252       double ct = t1;
253 
254       final ArrayList<byte[]> bufs = new ArrayList<byte[]>((int) Math.ceil(t2 - t1) + 1);
255       final String endDate = dateFormat.format(J2kSec.asDate(t2));
256       boolean done = false;
257       ResultSet rs = null;
258 
259       // MySQL only uses one key for optimization so selecting tracebufs
260       // that lie between st and et is prohibitively slow. By selecting
261       // the latest 1 tracebuf where the st is less than the desired time
262       // we can quickly find the tracebuf that may overlap into the
263       // desired interval. Unfortunately, this causes a problem for the
264       // edge case of selecting data right at the UTC day boundary. To
265       // fix this the query must be run against the day of the st of the
266       // desired interval plus the day before.
267       final String[] initialDates = new String[] {dateFormat.format(J2kSec.asDate(t1 - ONE_DAY)),
268           dateFormat.format(J2kSec.asDate(t1))};
269 
270       for (final String date : initialDates) {
271         final String sql = "SELECT st, et, sr, datatype, tracebuf FROM `" + code + "$$" + date
272             + "` WHERE st<? ORDER BY st DESC LIMIT 1";
273         final PreparedStatement ps = winston.getPreparedStatement(sql);
274         try {
275           ps.setDouble(1, t1);
276           rs = ps.executeQuery();
277           if (rs.next()) {
278             if (t1 >= rs.getDouble(1) && t1 <= rs.getDouble(2)) {
279               final byte[] buf = getTraceBufBytes(rs);
280               numSamplesCounter += getNumSamples(rs.getDouble(1), rs.getDouble(2), rs.getDouble(3));
281               if (maxrows > 0 && numSamplesCounter > maxrows) {
282                 throw new UtilException("Max rows (" + maxrows + " rows) "
283                     + (vdxName.length() > 0 ? ("for data source " + vdxName + " ") : "")
284                     + "exceeded.");
285               }
286               bufs.add(buf);
287             }
288           }
289           rs.close();
290         } catch (final SQLException e) {
291           LOGGER.debug("No table found for {}, {}->{}", date, t1, t2);
292         }
293       }
294 
295       // got the first tracebuf, now lets get the rest.
296       while (!done) {
297         final String date = dateFormat.format(J2kSec.asDate(ct));
298         if (date.equals(endDate))
299           done = true;
300 
301         final String sql = "SELECT st, et, sr, datatype, tracebuf FROM `" + code + "$$" + date
302             + "` WHERE st>=" + t1 + " AND st<=" + t2 + " ORDER BY st ASC";
303 
304         try {
305           rs = winston.getStatement().executeQuery(sql);
306         } catch (final SQLException e) {
307           // table not found
308           LOGGER.debug("No table found for {}$${})", code, date);
309           ct += ONE_DAY;
310           continue;
311         }
312         while (rs.next()) {
313 
314           final byte[] buf = getTraceBufBytes(rs);
315           numSamplesCounter += getNumSamples(rs.getDouble(1), rs.getDouble(2), rs.getDouble(3));
316           if (maxrows > 0 && numSamplesCounter > maxrows)
317             throw new UtilException("Max rows (" + maxrows + " rows) "
318                 + (vdxName.length() > 0 ? ("for data source " + vdxName + " ") : "") + "exceeded.");
319           bufs.add(buf);
320         }
321         rs.close();
322         ct += ONE_DAY;
323       }
324       return bufs;
325     } catch (final SQLException e) {
326       LOGGER.debug("Could not get TraceBuf bytes for {}, {}->{}", code, t1, t2);
327     } catch (final IOException e) {
328       LOGGER.debug("Could not get TraceBuf bytes for {}, {}->{}", code, t1, t2);
329     }
330     return null;
331 
332   }

353   public List<TraceBuf> getTraceBufs(final String code, final double t1, final double t2,
354       final int maxrows) throws UtilException {
355     final List<byte[]> rawBufs = getTraceBufBytes(code, t1, t2, maxrows);
356     if (rawBufs == null || rawBufs.size() == 0)
357       return null;
358 
359     try {
360       final List<TraceBuf> traceBufs = new ArrayList<TraceBuf>(rawBufs.size());
361       for (final byte[] buf : rawBufs)
362         traceBufs.add(new TraceBuf(buf));
363 
364       return traceBufs;
365     } catch (final Exception e) {
366       LOGGER.error("Could not get TraceBufs for {}, {}->{}", code, t1, t2);
367     }
368     return null;
369   }

372    * Return wave data for timespan t1..t2 for channel w/ code; cap result at
373    * maxrows size. Currently returns larger than asked for.
374    *
375    * @param code
376    *          channel
377    * @param t1
378    *          start time
379    * @param t2
380    *          end time
381    * @return wave
382    */
383   public Wave getWave(final String code, final double t1, final double t2, final int maxrows)
384       throws UtilException {
385     if (!winston.checkConnect() || !winston.useDatabase(code))
386       return null;
387 
388     final List<TraceBuf> bufs = getTraceBufs(code, t1, t2, maxrows);
389     if (bufs == null || bufs.size() == 0)
390       return null;
391     else {
392       final Wave wave = TraceBuf.traceBufToWave(bufs);
393       wave.convertToJ2K();
394       return wave;
395     }

398   public HelicorderData getHelicorderData(final String code, final double t1, final double t2,
399       final int maxrows) throws UtilException {
400     if (!winston.checkConnect() || !winston.useDatabase(code))
401       return null;
402     try {
403       double ct = t1;
404       // this 'fixes' problems when a start time of 0000 UTC is asked for
405       // and that data are actually stored in the previous day. Some
406       // issues remain.
407       final String endDate = dateFormat.format(J2kSec.asDate(t2));
408       boolean done = false;
409       final ArrayList<double[]> list = new ArrayList<double[]>((int) (t2 - t1) + 2);
410       while (!done) {
411         final String date = dateFormat.format(J2kSec.asDate(ct));
412         if (date.equals(endDate))
413           done = true;
414         ct += ONE_DAY;
415         final String table = code + "$$H" + date;
416         ResultSet rs = null;
417         String sql = "SELECT j2ksec, smin, smax, rcnt FROM `" + table
418             + "` WHERE j2ksec>=? AND j2ksec<=? ORDER BY j2ksec ASC";
419 
420         if (maxrows != 0) {
421           sql += " LIMIT " + (maxrows + 1);
422 
423           // Check the row count for the query before running the entire thing. Can save significant
424           // time
425           // for sufficiently large queries that exceed the maxrows parameter
426           final PreparedStatement s = winston.getPreparedStatement(
427               "SELECT COUNT(*) FROM (SELECT 1 " + sql.substring(sql.indexOf("FROM")) + ") as T");
428           try {
429             s.setDouble(1, t1);
430             s.setDouble(2, t2);
431             rs = s.executeQuery();
432           } catch (final Exception e) {
433             // table not found
434             continue;
435           }
436           if (rs.next() && rs.getInt(1) > maxrows)
437             throw new UtilException("Max rows (" + maxrows + " rows) "
438                 + (vdxName.length() > 0 ? ("for data source " + vdxName + " ") : "") + "exceeded.");
439         }
440 
441         final PreparedStatement select = winston.getPreparedStatement(sql);
442         try {
443           select.setDouble(1, t1);
444           select.setDouble(2, t2);
445           rs = select.executeQuery();
446         } catch (final Exception e) {
447           // table not found
448           continue;
449         }
450 
451         while (rs.next()) {
452           final double[] d = new double[] {rs.getDouble(1), rs.getDouble(2), rs.getDouble(3)}; // ,
453                                                                                                // rs.getInt(4)
454           // };
455           list.add(d);
456         }
457         rs.close();
458       }
459       return new HelicorderData(list);
460     } catch (final SQLException e) {
461       LOGGER.error("Could not get helicorder for {}, {}->{}", code, t1, t2);
462     }
463     return null;
464   }

466   public RSAMData getRSAMData(final String code, final double t1, final double t2,
467       final int maxrows, final DownsamplingType ds, final int dsInt) throws UtilException {
468     if (!winston.checkConnect() || !winston.useDatabase(code))
469       return null;
470 
471     try {
472       double ct = t1;
473       // this 'fixes' problems when a start time of 0000 UTC is asked for
474       // and that data are actually stored in the previous day. Some
475       // issues remain.
476       final String endDate = dateFormat.format(J2kSec.asDate(t2));
477       boolean done = false;
478       int numSamplesCounter = 0;
479       final ArrayList<double[]> list = new ArrayList<double[]>((int) (t2 - t1) + 2);
480       while (!done) {
481         final String date = dateFormat.format(J2kSec.asDate(ct));
482         if (date.equals(endDate))
483           done = true;
484         ct += ONE_DAY;
485         final String table = code + "$$H" + date;
486         ResultSet rs = null;
487         String sql = "SELECT j2ksec, rsam" + " FROM `" + table + "` WHERE j2ksec>=" + t1
488             + " AND j2ksec<=" + t2 + " AND rcnt>0" + " ORDER BY j2ksec";
489         try {
490           sql = getDownsamplingSQL(sql, t1, ds, dsInt);
491         } catch (final UtilException e) {
492           throw new UtilException("Can't downsample dataset: " + e.getMessage());
493         }
494         if (maxrows != 0) {
495           sql += " LIMIT " + (maxrows + 1);
496 
497           // If the dataset has a maxrows paramater, check that the number of requested rows doesn't
498           // exceed that number prior to running the full query. This can save a decent amount of
499           // time
500           // for large queries. Note that this only applies for non-downsampled queries. This is
501           // done for
502           // two reasons: 1) If the user is downsampling, they already know they're dealing with a
503           // lot of data
504           // and 2) the way MySQL handles the multiple nested queries that would result makes it
505           // slower than
506           // just doing the full query to begin with.
507           if (ds.equals(DownsamplingType.NONE)) {
508             try {
509               rs = winston.getStatement().executeQuery("SELECT COUNT(*) FROM (SELECT 1 "
510                   + sql.substring(sql.indexOf("FROM")) + ") as T");
511             } catch (final Exception e) {
512               // table not found
513               continue;
514             }
515             if (rs.next() && rs.getInt(1) > maxrows) {
516               throw new UtilException("Max rows (" + maxrows + " rows) "
517                   + (vdxName.length() > 0 ? ("for data source " + vdxName + " ") : "")
518                   + "exceeded.");
519             }
520           }
521         }
522 
523         try {
524           rs = winston.getStatement().executeQuery(sql);
525         } catch (final Exception e) {
526           // table not found
527           continue;
528         }
529 
530         // Check for the amount of data returned in a downsampled query. Non-downsampled queries are
531         // checked above.
532         if (!ds.equals(DownsamplingType.NONE) && maxrows != 0) {
533           numSamplesCounter += getResultSetSize(rs);
534           if (numSamplesCounter > maxrows) {
535             throw new UtilException("Max rows (" + maxrows + " rows) "
536                 + (vdxName.length() > 0 ? ("for data source " + vdxName + " ") : "") + "exceeded.");
537           }
538         }
539         while (rs.next()) {
540           final double[] d = new double[] {rs.getDouble(1), rs.getDouble(2)};
541           list.add(d);
542         }
543         rs.close();
544       }
545       return new RSAMData(list);
546     } catch (final SQLException e) {
547       LOGGER.error("Could not get RSAM for {}, {}->{}", code, t1, t2);
548     }
549     return null;
550   }

552   /**
553    * Version of SQLDataSource.getDownsamplingSQL which doesn't use prepared statements, so as it's
554    * not supported by
555    * Winston.
556    */
557   private static String getDownsamplingSQL(final String sql, final double startTime,
558       final DownsamplingType ds, final int dsInt) throws UtilException {
559     if (!ds.equals(DownsamplingType.NONE) && dsInt <= 1)
560       throw new UtilException("Downsampling interval should be more than 1");
561     if (ds.equals(DownsamplingType.NONE))
562       return sql;
563     else if (ds.equals(DownsamplingType.DECIMATE))
564       return "SELECT * FROM(SELECT fullQuery.*, @row := @row+1 AS rownum FROM (" + sql
565           + ") fullQuery, (SELECT @row:=0) r) ranked WHERE rownum % " + dsInt + " = 1";
566     else if (ds.equals(DownsamplingType.MEAN)) {
567       final String sql_select_clause = sql.substring(6, sql.toUpperCase().indexOf("FROM") - 1);
568       final String sql_from_where_clause = sql.substring(sql.toUpperCase().indexOf("FROM") - 1,
569           sql.toUpperCase().lastIndexOf("ORDER BY") - 1);
570       final String[] columns = sql_select_clause.split(",");
571       String avg_sql = "SELECT ";
572       for (final String column : columns) {
573         avg_sql += "AVG(" + column.trim() + "), ";
574       }
575       avg_sql += "((j2ksec-" + startTime + ") DIV " + dsInt + ") intNum ";
576       avg_sql += sql_from_where_clause;
577       avg_sql += " GROUP BY intNum";
578       return avg_sql;
579     } else
580       throw new UtilException("Unknown downsampling type: " + ds);
581   }

587   public static int getResultSetSize(final ResultSet rs) throws SQLException {
588     if (rs == null) {
589       return 0;
590     }
591 
592     final int currentRow = rs.getRow();
593     rs.beforeFirst();
594     rs.last();
595     final int size = rs.getRow();
596     if (currentRow == 0) {
597       rs.beforeFirst();
598     } else {
599       rs.absolute(currentRow);
600     }
601     return size;
602   }

25  /**
26   * A base class for writing a static data importer for Winston. Clients just
27   * have to fill the instructions <code>StringBuffer</code>, implement
28   * <code>readFile()</code> and call <code>process()</code>.
29   *
30   * <code>readFile()</code> is responsible for creating a map that maps Winston
31   * compatible channel codes to lists of TRACEBUF-sized <code>Wave</code>s.
32   *
33   *
34   * @author Dan Cervelli
35   */
36  abstract public class StaticImporter {
37    protected static StringBuffer instructions = new StringBuffer();
38  
39    protected WinstonDatabase winston;
40    protected InputEW input;
41    protected Channels channels;
42    protected Data data;
43  
44    protected static String driver;
45    protected static String url;
46    protected static String db;
47  
48    protected boolean rsamEnable = true;
49    protected int rsamDelta = 10;
50    protected int rsamDuration = 60;
51  
52    public StaticImporter() {}
53  
54    public void setupWinston() {
55      final ConfigFile cf = new ConfigFile("Winston.config");
56      final String driver = cf.getString("winston.driver");
57      final String url = cf.getString("winston.url");
58      final String db = cf.getString("winston.prefix");
59      winston = new WinstonDatabase(driver, url, db);
60      if (!winston.checkDatabase()) {
61        System.err.println("Winston database does not exist and could not be created.");
62        System.exit(-1);
63      }
64      input = new InputEW(winston);
65      channels = new Channels(winston);
66      data = new Data(winston);
67    }
68  
69    public void importMap(final Map<String, List<Wave>> map) {
70      if (map == null) {
71        System.out.println("Nothing to import.");
72        return;
73      }
74  
75      final CodeTimer timer = new CodeTimer("import");
76      double minTime = 1E300;
77      double maxTime = -1E300;
78      for (final String code : map.keySet()) {
79        if (!channels.channelExists(code)) {
80          System.out.println("Creating new channel '" + code + "' in Winston.");
81          channels.createChannel(code);
82        }
83  
84        System.out.printf("Importing channel: %s.\n", code);
85        List<Wave> waves = map.get(code);
86        System.out.printf("Converting %d waves into TraceBufs.\n", waves.size());
87        final List<TraceBuf> tbs = new ArrayList<TraceBuf>(waves.size());
88  
89        final int maxSamples = (int) (Math.pow(2, 16) / 4);
90        final Iterator<Wave> it = waves.iterator();
91        final List<Wave> subWaves = new ArrayList<Wave>();
92        while (it.hasNext()) {
93          final Wave wave = it.next();
94          if (wave.numSamples() > maxSamples) {
95            subWaves.addAll(wave.split(maxSamples));
96            it.remove();
97          }
98        }
99        waves.addAll(subWaves);
100 
101       for (Wave wave : waves) {
102         final TraceBuf tb = new TraceBuf(code, wave);
103         minTime = Math.min(minTime, tb.firstSampleTime());
104         maxTime = Math.max(maxTime, tb.lastSampleTime());
105         tb.createBytes();
106         tbs.add(tb);
107         wave = null; // for garbage collection
108       }
109       waves = null;
110 
111       final int duration = (int) (maxTime - minTime);
112       input.setRowParameters(duration + 5, Math.min(duration, 100));
113       System.out.println("Writing TraceBufs to database.");
114       input.inputTraceBufs(tbs, rsamEnable, rsamDelta, rsamDuration);
115 
116       System.out.println("Done.");
117     }
118     timer.stop();
119     System.out.printf("Completed in %.2fs\n", timer.getRunTimeMillis() / 1000);
120   }

122   public static void process(final List<String> files, final StaticImporter impt) {
123     if (files.size() == 0) {
124       System.out.println("No files to import.");
125       System.exit(1);
126     }
127     impt.setupWinston();
128 
129     final ListIterator<String> it = files.listIterator();
130     while (it.hasNext()) {
131       final String fn = it.next();
132       System.out.println("Reading file: " + fn);
133 
134       // assume files really means files, not resource records --tjp
135       final File f = new File(fn);
136       if (f.isDirectory()) {
137         File[] dFiles = f.listFiles();
138         if (dFiles == null)
139           continue;
140         for (final File ff : dFiles) {
141           
142           if (ff.getName() != null)
143             it.add(fn + File.pathSeparatorChar + ff.getName());
144         }
145       } else {
146         try {
147           impt.importMap(impt.readFile(fn));
148         } catch (final IOException e) {
149           System.out.println("Can't read " + fn + " (" + e.getLocalizedMessage() + ")");
150         }
151       }
152     }
153 
154   }

163   public void processArguments(final Arguments args) {
164     final String rd = args.get("-rd");
165     rsamDelta = StringUtils.stringToInt(rd, 10);
166     final String rl = args.get("-rl");
167     rsamDuration = StringUtils.stringToInt(rl, 60);
168     System.out.printf("RSAM parameters: delta=%d, duration=%d.\n", rsamDelta, rsamDuration);
169   }

137 	public void processConfig()
138 	{
139 		createDatabase = Util.stringToBoolean(config.getString("createDatabase"));
140 		logger.fine("createDatabase: " + createDatabase);
141 		
142 		winston = WinstonDatabase.processWinstonConfigFile(config);
143 		logger.fine("winston.driver: " + winston.dbDriver);
144 		logger.fine("winston.url: " + winston.dbURL);
145 		logger.fine("winston.prefix: " + winston.databasePrefix);
146 		logger.fine("winston.statementCacheCap: " + winston.cacheCap);
147 		
148 		if (createDatabase)
149 		{
150 			winston.checkDatabase();
151 		}
152 		
153 		String s = config.getString("waveServer");
154 		if (s == null)
155 			throw new RuntimeException("no waveServer string");
156 		
157 		waveServer = new WaveServer(config.getString("waveServer"));
158 		logger.fine("waveServer: " + waveServer.host + ":" + waveServer.port);
159 		
160 		createChannels = Util.stringToBoolean(config.getString("createChannels"));
161 		logger.fine("createChannels: " + createChannels);
162 		
163 		sourceChannels = config.getList("channel");
164 		logger.fine("sourceChannels: " + sourceChannels);
165 		
166 		String timeRange = config.getString("timeRange");
167 		logger.fine("timeRange: " + timeRange);
168 		parseTimeRange(timeRange);
169 		
170 		chunkSize = Util.stringToDouble(config.getString("chunkSize"), DEFAULT_CHUNK_SIZE);
171 		logger.fine("chunkSize: " + chunkSize);
172 		
173 		chunkDelay = Util.stringToInt(config.getString("chunkDelay"), DEFAULT_CHUNK_DELAY);
174 		logger.fine("chunkDelay: " + chunkDelay);
175 	
176 		rsamEnable = Util.stringToBoolean(config.getString("rsam.enable"), DEFAULT_RSAM_ENABLE);
177 		logger.fine("rsamEnable: " + rsamEnable);
178 		
179 		rsamDelta = Util.stringToInt(config.getString("rsam.delta"), DEFAULT_RSAM_DELTA);
180 		logger.fine("rsamDelta: " + rsamDelta);
181 		
182 		rsamDuration = Util.stringToInt(config.getString("rsam.duration"), DEFAULT_RSAM_DURATION);
183 		logger.fine("rsamDuration: " + rsamDuration);
184 		// TODO: log level
185 	}

194 	private void parseTimeRange(String timeRange)
195 	{		
196 		try
197 		{
198 			double[] tr = Time.parseTimeRange(timeRange);
199 			startTime = tr[0];
200 			endTime = tr[1];
201 		}
202 		catch (Exception e)
203 		{
204 			logger.severe("Error parsing time range: " + e.getMessage());
205 			System.exit(-1);
206 		}
207 		
208 		logger.fine(String.format("Requested time range: [%s -> %s, %s]",
209 				Time.toDateString(startTime),
210 				Time.toDateString(endTime), 
211 				Util.timeDifferenceToString(endTime - startTime)));
212 	}

238 	public void createJobs()
239 	{
240 		getChannels();
241 		jobs = new ArrayList<ImportWSJob>();
242 		
243 		List<MenuItem> items = menu.getItems();
244 		for (MenuItem item : items)
245 		{
246 			for (String channel : sourceChannels)
247 			{
248 				String[] ss = channel.split("[\\$\\_ ]");
249 				String loc = null;
250 				if (ss.length == 4)
251 					loc = ss[3];
252 				
253 				if (item.match(ss[0], ss[1], ss[2], loc))
254 				{
255 					String wc = item.getSCNSCNL("$");
256 					
257 					if (!createChannels && !channels.channelExists(wc))
258 						continue;
259 					
260 					logger.fine("Remote channel matched: " + wc);
261 					ImportWSJob job = new ImportWSJob(winston, waveServer, this);
262 					job.setChannel(wc);
263 					job.setChunkDelay(chunkDelay);
264 					job.setChunkSize(chunkSize);
265 					job.setRSAMParameters(rsamEnable, rsamDelta, rsamDuration);
266 					jobs.add(job);
267 				}
268 			}
269 		}
270 	}

272 	public void startImport()
273 	{
274 		for (ImportWSJob job : jobs)
275 		{
276 			currentJob = job;
277 			logger.info(job.getChannel() + ": finding gaps");
278 			List<double[]> gaps = data.findGaps(job.getChannel(), startTime, endTime);
279 			for (double[] gap : gaps)
280 				job.addSpan(gap[0], gap[1]);
281 
282 			job.go();
283 			
284 			if (quit)
285 				break;
286 		}
287 		appTimer.stop();
288 		logger.info(String.format("%d tbs inserted, total download time: %s, total insert time: %s (%.3fms/tb)", 
289 				totalInserted, Util.timeDifferenceToString(totalDownloadTime / 1000), Util.timeDifferenceToString(totalInsertTime / 1000),
290 				totalInsertTime / (double)totalInserted));
291 		logger.info("Total run time: " + Util.timeDifferenceToString(appTimer.getRunTimeMillis() / 1000.0));
292 		quit = true;
293 	}

296 	{
297 		Thread launchThread = new Thread(new Runnable()
298 				{
299 					public void run()
300 					{

317 	/**
318 	 * Find and parse the command line arguments.
319 	 *
320 	 * @param args The command line arguments.
321 	 */
322 	public static JSAPResult getArguments(String[] args) {
323 		JSAPResult config = null;
324 		try
325 		{
326 			SimpleJSAP jsap = new SimpleJSAP(
327 					JSAP_PROGRAM_NAME,
328 					JSAP_EXPLANATION_PREFACE + DEFAULT_JSAP_EXPLANATION,
329 					DEFAULT_JSAP_PARAMETERS
330 					);
331 			
332 			config = jsap.parse(args);
333 			
334 			if (jsap.messagePrinted())
335 			{
336 				// The following error message is useful for catching the case
337 				// when args are missing, but help isn't printed.
338 				if (!config.getBoolean("help"))
339 					System.err.println("Try using the --help flag.");
340 
341 				System.exit(1);
342 			}
343 		}
344 		catch (Exception ex)
345 		{
346 			ex.printStackTrace();
347 			System.exit(1);
348 		}
349 		return config;
350 	}

356 	public static void main(String[] args) throws IOException
357 	{
358 		JSAPResult config = getArguments(args);
359 		ImportWS w = new ImportWS(config.getString("configFilename"));
360 
361 		if (config.getString("timeRange") != null)
362 			w.parseTimeRange(config.getString("timeRange"));		
363 
364 		if (config.getString("waveServer") != null)
365 			w.waveServer = new WaveServer(config.getString("waveServer"));		
366 		
367 		boolean acceptCommands = !(config.getBoolean("noInput"));
368 
369 		w.setRequestSCNL((config.getBoolean("SCNL")));
370 		
371 		w.createJobs();
372 		w.go();
373 		BufferedReader in = new BufferedReader(new InputStreamReader(System.in));
374 				
375 		while (acceptCommands && !w.quit)
376 		{
377 			String s = in.readLine();
378 			if (s != null)
379 			{
380 				s = s.toLowerCase().trim();
381 				if (s.equals("q"))
382 					w.quit();
383 			}
384 		}
385 	}

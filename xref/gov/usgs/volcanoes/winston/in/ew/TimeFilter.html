1   package gov.usgs.volcanoes.winston.in.ew;
2   
3   import gov.usgs.earthworm.message.TraceBuf;
4   import gov.usgs.volcanoes.core.configfile.ConfigFile;
5   import gov.usgs.volcanoes.core.time.J2kSec;
6   import gov.usgs.volcanoes.core.util.StringUtils;
7   
8   /**
9    * A TraceBuf filter that rejects packets based on the difference between
10   * their time and the current time.
11   *
12   * $Log: not supported by cvs2svn $
13   * 
14   * @author Dan Cervelli
15   */
16  public class TimeFilter extends TraceBufFilter {
17    private double tLessThan;
18    private double tGreaterThan;
19  
20    public TimeFilter() {}
21  
22    public TimeFilter(final double tlt, final double tgt) {
23      this();
24      tLessThan = tlt;
25      tGreaterThan = tgt;
26    }
27  
28    @Override
29    public void configure(final ConfigFile cf) {
30      super.configure(cf);
31      if (cf == null)
32        return;
33  
34      tLessThan = StringUtils.stringToDouble(cf.getString("past"), Double.NaN);
35      tGreaterThan = StringUtils.stringToDouble(cf.getString("future"), Double.NaN);
36    }
37  
38    @Override
39    public boolean match(final TraceBuf tb, final Options options) {
40      final double dt = tb.getStartTimeJ2K() - J2kSec.now();
41      if (!Double.isNaN(tLessThan) && dt < tLessThan)
42        return true;
43  
44      if (!Double.isNaN(tGreaterThan) && dt > tGreaterThan)
45        return true;
46  
47      return false;
48    }
49  
50    @Override
51    public String toString() {
52      return String.format("TimeFilter [%s: %.2f > dt > %.2f]", accept ? "accept" : "reject",
53          tLessThan, tGreaterThan);
54    }
55  }

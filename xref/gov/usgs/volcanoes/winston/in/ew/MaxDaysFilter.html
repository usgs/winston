1   package gov.usgs.volcanoes.winston.in.ew;

3   import gov.usgs.earthworm.message.TraceBuf;
4   import gov.usgs.volcanoes.core.time.J2kSec;
5   
6   /**
7    *
8    * $Log: not supported by cvs2svn $
9    * 
10   * @author Dan Cervelli
11   */
12  public class MaxDaysFilter extends TraceBufFilter {
13    public MaxDaysFilter() {}
14  
15    @Override
16    public boolean match(final TraceBuf tb, final Options options) {
17      if (options.maxDays <= 0)
18        return false;
19  
20      final double t = tb.getStartTimeJ2K();
21      final double ds = J2kSec.now() - t;
22      return (ds > (options.maxDays * 86400.0));
23    }
24  
25    @Override
26    public String toString() {
27      return "MaxDaysFilter";
28    }
29  }

3   import gov.usgs.util.ConfigFile;
4   import gov.usgs.util.CurrentTime;
5   import gov.usgs.util.Util;
6   
7   /**
8    * 
9    * $Log: not supported by cvs2svn $
10   * @author Dan Cervelli
11   */
12  public class Options
13  {
14  	public double timeThreshold;
15  	public int bufThreshold;
16  	public int maxBacklog;
17  	public boolean rsamEnable;
18  	public int rsamDelta;
19  	public int rsamDuration;
20  	public int maxDays;
21  	
22  	public Options()
23  	{}
24  	
25  	public static Options createOptions(ConfigFile cf, Options defaults)
26  	{
27  		Options threshold = new Options();
28  		threshold.timeThreshold = Util.stringToDouble(cf.getString("timeThreshold"), 
29  				defaults.timeThreshold);
30  		threshold.bufThreshold = Util.stringToInt(cf.getString("traceBufThreshold"), 
31  				defaults.bufThreshold);
32  		threshold.maxBacklog = Util.stringToInt(cf.getString("maxBacklog"), 
33  				defaults.maxBacklog);
34  		threshold.rsamEnable = Util.stringToBoolean(cf.getString("rsam.enable"),
35  				defaults.rsamEnable);
36  		threshold.rsamDelta = Util.stringToInt(cf.getString("rsam.delta"), 
37  				defaults.rsamDelta);
38  		threshold.rsamDuration = Util.stringToInt(cf.getString("rsam.duration"), 
39  				defaults.rsamDuration);
40  		threshold.maxDays = Util.stringToInt(cf.getString("maxDays"), 
41  				defaults.maxDays);
42  		return threshold;
43  	}
44  	
45  	public boolean thresholdExceeded(double time, int size)
46  	{
47  		if (timeThreshold != -1)
48  		{
49  			double dt = CurrentTime.getInstance().nowJ2K() - time;
50  			if (dt > timeThreshold)
51  				return true;
52  		}
53  		
54  		if (size > bufThreshold)
55  			return true;
56  		
57  		return false;
58  	}
59  	
60  	public String toString()
61  	{
62  		return String.format("timeThreshold=%.1f, traceBufThreshold=%d, maxBacklog=%d, maxDays=%d, rsam.enable=%s, rsam.delta=%d, rsam.duration=%d",
63  				timeThreshold, bufThreshold, maxBacklog, maxDays, rsamEnable, rsamDelta, rsamDuration);
64  	}
65  }

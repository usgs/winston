7   import gov.usgs.net.NetTools;
8   import gov.usgs.volcanoes.winston.db.WinstonDatabase;
9   import gov.usgs.volcanoes.winston.server.WWS;
10  import gov.usgs.volcanoes.winston.server.WWSCommandString;
11  
12  /**
13   * example command: MENU -
14   *
15   * @author Dan Cervelli
16   */
17  public class MenuCommand extends BaseCommand {
18    public MenuCommand(final NetTools nt, final WinstonDatabase db, final WWS wws) {
19      super(nt, db, wws);
20    }
21  
22    public void doCommand(final Object info, final SocketChannel channel) {
23      final WWSCommandString cmd = new WWSCommandString((String) info);
24      boolean scnl = false;
25      if (cmd.length() == 3 && cmd.getString(2).equals("SCNL"))
26        scnl = true;
27  
28      if (!scnl && !cmd.isLegal(2))
29        return; // malformed command;
30  
31      final StringBuffer sb = new StringBuffer(4096);
32      sb.append(cmd.getID() + " ");
33      final List<String> menu = emulator.getWaveServerMenu(scnl, 0, 0, maxDays);
34      for (final String s : menu)
35        sb.append(s);
36      sb.append('\n');
37  
38      netTools.writeString(sb.toString(), channel);
39      wws.log(Level.FINER, "MENU" + (scnl ? " (SCNL)" : "") + ".", channel);
40    }
41  }

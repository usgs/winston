7   package gov.usgs.volcanoes.winston.server;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import gov.usgs.volcanoes.core.configfile.ConfigFile;
13  import gov.usgs.volcanoes.winston.server.wwsCmd.UnsupportedCommandException;
14  import gov.usgs.volcanoes.winston.server.wwsCmd.WwsBaseCommand;
15  import gov.usgs.volcanoes.winston.server.wwsCmd.WwsCommand;
16  import gov.usgs.volcanoes.winston.server.wwsCmd.WwsCommandString;
17  import io.netty.buffer.Unpooled;
18  import io.netty.channel.ChannelFutureListener;
19  import io.netty.channel.ChannelHandlerContext;
20  import io.netty.channel.SimpleChannelInboundHandler;
21  
22  public class WwsServerHandler extends SimpleChannelInboundHandler<WwsCommandString> {
23  
24    private static final Logger LOGGER = LoggerFactory.getLogger(WwsServerHandler.class);
25  
26    private final WinstonDatabasePool winstonDatabasePool;
27  
28    public WwsServerHandler(ConfigFile configFile, WinstonDatabasePool winstonDatabasePool) {
29      this.winstonDatabasePool = winstonDatabasePool;
30    }
31  
32    @Override
33    public void channelRead0(ChannelHandlerContext ctx, WwsCommandString request) throws Exception {
34  
35      try {
36        final WwsBaseCommand wwsWorker = WwsCommand.get(winstonDatabasePool, request);
37        wwsWorker.doCommand(ctx, request);
38      } catch (final UnsupportedCommandException e) {
39        LOGGER.info(e.getLocalizedMessage());
40        ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(ChannelFutureListener.CLOSE);
41      }
42    }
43  
44    @Override
45    public void channelReadComplete(ChannelHandlerContext ctx) {
46      ctx.flush();
47    }
48  
49    @Override
50    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
51      LOGGER.error("Exception caught in WwsServerHandler.", cause);
52      ctx.close();
53    }
54  }

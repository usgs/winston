6   package gov.usgs.volcanoes.winston.server.http.cmd.fdsnws;
7   
8   import java.text.ParseException;
9   import java.text.SimpleDateFormat;
10  import java.util.Date;
11  
12  /**
13   * Format and parse a date. Should this subclass SimpleDateFormatter?
14   * 
15   * @author Tom Parker
16   *
17   */
18  public class FdsnwsDate {
19    private static String DATE_STRING = "YYYY-MM-DD'T'HH:MM:SS.ss";
20    
21    /**
22     * Format a date.
23     * 
24     * @param date date
25     * @return string representation
26     */
27    public static String toString(Date date) {
28      SimpleDateFormat sFormat = new SimpleDateFormat(DATE_STRING);
29      return sFormat.format(date);
30    }
31    
32    /**
33     * Parse a date.
34     * 
35     * @param date string date
36     * @return date object
37     * @throws ParseException when provided a malformed string
38     */
39    public static Date parse(String date) throws ParseException {
40      SimpleDateFormat sFormat = new SimpleDateFormat(DATE_STRING);
41      return sFormat.parse(date);
42    }
43  }

6   package gov.usgs.volcanoes.winston.server.wws.cmd;
7   
8   import java.util.ArrayList;
9   import java.util.Arrays;
10  import java.util.List;
11  
12  import gov.usgs.volcanoes.core.time.J2kSec;
13  import gov.usgs.volcanoes.core.util.StringUtils;
14  import gov.usgs.volcanoes.core.util.UtilException;
15  import gov.usgs.volcanoes.winston.Channel;
16  import gov.usgs.volcanoes.winston.db.Channels;
17  import gov.usgs.volcanoes.winston.db.WinstonDatabase;
18  import gov.usgs.volcanoes.winston.server.ConnectionStatistics;
19  import gov.usgs.volcanoes.winston.server.MalformedCommandException;
20  import gov.usgs.volcanoes.winston.server.wws.WinstonConsumer;
21  import gov.usgs.volcanoes.winston.server.wws.WwsBaseCommand;
22  import gov.usgs.volcanoes.winston.server.wws.WwsCommandString;
23  import io.netty.channel.ChannelHandlerContext;
24  import io.netty.util.AttributeKey;
25  
26  public class StatusCommand extends WwsBaseCommand {
27  
28    private ConnectionStatistics connectionStatistics;
29  
30    private static final AttributeKey<ConnectionStatistics> connectionStatsKey;
31  
32    static {
33      connectionStatsKey = AttributeKey.valueOf("connectionStatistics");
34    }
35  
36    public StatusCommand() {
37      super();
38    }
39  
40    public void doCommand(ChannelHandlerContext ctx, WwsCommandString cmd)
41        throws MalformedCommandException, UtilException {
42  
43      final double ageThreshold = StringUtils.stringToDouble(cmd.getString(2), 0);
44      final double now = J2kSec.fromEpoch(System.currentTimeMillis());
45  
46      final StringBuilder sb = new StringBuilder();
47      int lines = 0;
48  
49      connectionStatistics = ctx.channel().attr(connectionStatsKey).get();
50      sb.append(String.format("Connection count: %d\n", connectionStatistics.getCount()));
51      lines++;
52  
53      List<Channel> sts;
54      try {
55        sts = databasePool.doCommand(new WinstonConsumer<List<Channel>>() {
56          public List<Channel> execute(WinstonDatabase winston) {
57            return new Channels(winston).getChannels();
58          }
59        });
60      } catch (Exception e) {
61        throw new UtilException("Unable to get channels for status command");
62      }
63  
64      sb.append(String.format("Channel count: %d\n", sts.size()));
65      lines++;
66  
67      final ArrayList<Double> ages = new ArrayList<Double>();
68      for (final Channel st : sts)
69        if (st.getMaxTime() < now && (ageThreshold == 0 || now - st.getMaxTime() < ageThreshold))
70          ages.add(now - st.getMaxTime());
71  
72      if (ages.size() == 0)
73        ages.add(0d);
74  
75      Double[] d = new Double[ages.size() - 1];
76      d = ages.toArray(d);
77      Arrays.sort(d);
78  
79      sb.append(String.format("Median data age: %s\n", d[(d.length - 1) / 2]));
80      lines++;
81  
82      ctx.write("GC: " + lines + '\n');
83      ctx.writeAndFlush(sb.toString());
84    }
85  }

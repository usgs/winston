7   package gov.usgs.volcanoes.winston.server.wws;
8   
9   import org.slf4j.Logger;
10  import org.slf4j.LoggerFactory;
11  
12  import gov.usgs.volcanoes.core.configfile.ConfigFile;
13  import gov.usgs.volcanoes.winston.server.ConnectionStatistics;
14  import gov.usgs.volcanoes.winston.server.WinstonDatabasePool;
15  import io.netty.buffer.Unpooled;
16  import io.netty.channel.ChannelFutureListener;
17  import io.netty.channel.ChannelHandlerContext;
18  import io.netty.channel.SimpleChannelInboundHandler;
19  import io.netty.util.AttributeKey;
20  
21  /**
22   * 
23   * 
24   * @author Tom Parker
25   *
26   */
27  public class WwsCommandHandler extends SimpleChannelInboundHandler<WwsCommandString> {
28  
29    private static final Logger LOGGER = LoggerFactory.getLogger(WwsCommandHandler.class);
30    private static final int DEFAULT_MAX_DAYS = 20000;
31  
32    private final WinstonDatabasePool winstonDatabasePool;
33    private final ConfigFile configFile;
34    private ConnectionStatistics connectionStatistics;
35  
36    private static final AttributeKey<ConnectionStatistics> connectionStatsKey;
37  
38    static {
39      connectionStatsKey = AttributeKey.valueOf("connectionStatistics");
40    }
41  
42    /**
43     * Constructor.
44     * 
45     * @param configFile my config file
46     * @param winstonDatabasePool my database pool
47     */
48    public WwsCommandHandler(ConfigFile configFile, WinstonDatabasePool winstonDatabasePool) {
49      this.winstonDatabasePool = winstonDatabasePool;
50      this.configFile = configFile;
51    }
52  
53    @Override
54    public void channelRead0(ChannelHandlerContext ctx, WwsCommandString request) throws Exception {
55      connectionStatistics = ctx.channel().attr(connectionStatsKey).get();
56  
57      try {
58        final WwsBaseCommand wwsWorker = WwsCommandFactory.get(winstonDatabasePool, request);
59        connectionStatistics.incrWwsCount(ctx.channel().remoteAddress());
60        int maxDays = configFile.getInt("wws.maxDays", DEFAULT_MAX_DAYS);
61        if (maxDays == 0) {
62          maxDays = DEFAULT_MAX_DAYS;
63        }
64        wwsWorker.setMaxDays(maxDays);
65        wwsWorker.respond(ctx, request);
66      } catch (final UnsupportedCommandException e) {
67        LOGGER.info(e.getLocalizedMessage());
68        ctx.writeAndFlush(Unpooled.EMPTY_BUFFER).addListener(ChannelFutureListener.CLOSE);
69      }
70    }
71  
72    @Override
73    public void channelReadComplete(ChannelHandlerContext ctx) {
74      ctx.flush();
75    }
76  
77    @Override
78    public void exceptionCaught(ChannelHandlerContext ctx, Throwable cause) throws Exception {
79      LOGGER.error("Exception caught in WwsCommandHandler.", cause);
80      ctx.close();
81    }
82  }

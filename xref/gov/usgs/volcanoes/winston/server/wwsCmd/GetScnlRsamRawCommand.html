42   * @author Dan Cervelli

44   */
45  public class GetScnlRsamRawCommand extends WwsBaseCommand {
46    private static final Logger LOGGER = LoggerFactory.getLogger(GetScnlRsamRawCommand.class);
47  
48    public GetScnlRsamRawCommand() {
49      super();
50    }
51  
52    public void doCommand(ChannelHandlerContext ctx, WwsCommandString cmd)
53        throws MalformedCommandException, UtilException {
54  
55      if (!cmd.isLegalSCNLTT(10) || Double.isNaN(cmd.getDouble(8))
56          || cmd.getInt(9) == Integer.MIN_VALUE) {
57        throw new MalformedCommandException();
58      }
59      final double t1 = cmd.getT1(true);
60      final double t2 = cmd.getT2(true);
61      final int ds = (int) cmd.getDouble(8);
62      final String scnl = cmd.getWinstonSCNL();
63      final DownsamplingType dst = (ds < 2) ? DownsamplingType.NONE : DownsamplingType.MEAN;
64  
65      RSAMData rsam;
66      try {
67        rsam = databasePool.doCommand(new WinstonConsumer<RSAMData>() {
68          public RSAMData execute(WinstonDatabase winston) throws UtilException {
69            return new Data(winston).getRSAMData(scnl, t1, t2, 0, dst, ds);
70          }
71  
72        });
73      } catch (Exception e) {
74        throw new UtilException(e.getMessage());
75      }
76  
77      ByteBuffer bb = null;
78      if (rsam != null && rsam.rows() > 0)
79        bb = (ByteBuffer) rsam.toBinary().flip();
80      
81      if (cmd.getInt(9) == 1)
82        bb = ByteBuffer.wrap(Zip.compress(bb.array()));
83  
84      LOGGER.warn("returning {} rsam bytes", bb.limit());
85      ctx.write(cmd.getID() + " " + bb.limit() + '\n');
86      ctx.writeAndFlush(bb.array());
87    }
88  }

20  public class Instrument {
21    public static Instrument NULL = new Instrument();
22  
23    private int iid;
24    private String name;
25    private String description;
26    private double longitude = -999;
27    private double latitude = -999;
28    private double height = -999;
29  
30    private String timeZone;
31  
32    private Map<String, String> metadata;
33  
34    public Instrument() {}
35  
36    public Instrument(final ResultSet rs) throws SQLException {
37      iid = rs.getInt("instruments.iid");
38      name = rs.getString("name");
39      description = rs.getString("description");
40      longitude = rs.getDouble("lon");
41      boolean nll = rs.wasNull();
42      latitude = rs.getDouble("lat");
43      nll = nll || rs.wasNull();
44      if (nll) {
45        longitude = -999;
46        latitude = -999;
47      }
48      height = rs.getDouble("height");
49      timeZone = rs.getString("timezone");
50    }
51  
52    public int getID() {
53      return iid;
54    }
55  
56    public void setMetadata(final Map<String, String> map) {
57      metadata = map;
58    }
59  
60    public Map<String, String> getMetadata() {
61      return metadata;
62    }
63  
64    public void setLongitude(final double d) {
65      longitude = d;
66    }
67  
68    public void setLatitude(final double d) {
69      latitude = d;
70    }
71  
72    public double getLongitude() {
73      if (longitude == -999)
74        return Double.NaN;
75      return longitude;
76    }
77  
78    public double getLatitude() {
79      if (latitude == -999)
80        return Double.NaN;
81      return latitude;
82    }
83  
84    public double getHeight() {
85      if (height == -999)
86        return Double.NaN;
87      return height;
88    }
89  
90    public void setTimeZone(final String s) {
91      timeZone = s;
92    }
93  
94    public String getTimeZone() {
95      return timeZone == null ? "" : timeZone;
96    }
97  
98    public String getName() {
99      return name;
100   }
101 
102   public String getDescription() {
103     return description;
104   }
105

107   public String toString() {

109   }

124     this.iid = iid;

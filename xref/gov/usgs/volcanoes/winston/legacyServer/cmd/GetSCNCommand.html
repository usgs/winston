19  public class GetSCNCommand extends BaseCommand {
20  
21    public GetSCNCommand(final NetTools nt, final WinstonDatabase db, final WWS wws) {
22      super(nt, db, wws);
23    }
24  
25    public void doCommand(final Object info, final SocketChannel channel) {
26      int totalBytes = 0;
27      final WWSCommandString cmd = new WWSCommandString((String) info);
28      if (!cmd.isLegalSCNTT(8))
29        return;
30  
31      final int sid = emulator.getChannelID(cmd.getS(), cmd.getC(), cmd.getN());
32      if (sid == -1) {
33        sendNoChannelResponse(cmd.getID(), 0, cmd.getS(), cmd.getC(), cmd.getN(), null, channel);
34        return;
35      }
36  
37      double t1 = Time.ewToj2k(cmd.getT1(false));
38      t1 = timeOrMaxDays(t1);
39  
40      double t2 = Time.ewToj2k(cmd.getT2(false));
41      t2 = timeOrMaxDays(t2);
42  
43      final double[] bounds = checkTimes(sid, t1, t2);
44      if (!allowTransaction(bounds)) {
45        netTools.writeString(cmd.getEarthwormErrorString(sid, getError(bounds)), channel);
46        return;
47      }
48      Wave wave = null;
49      try {
50        wave = data.getWave(sid, t1, t2, 0);
51      } catch (final UtilException e) {
52      }
53      if (wave != null)
54        totalBytes = writeWaveAsAscii(wave, sid, cmd.getID(), cmd.getS(), cmd.getC(), cmd.getN(),
55            null, t1, t2, cmd.getString(7), channel);
56      else
57        netTools.writeString(cmd.getEarthwormErrorString(sid, "FG s4"), channel);
58  
59      final String scn = cmd.getS() + "_" + cmd.getC() + "_" + cmd.getN();
60      final String time = J2kSec.toDateString(t1) + " - " + J2kSec.toDateString(t2);
61      wws.log(Level.FINER, "GETSCN " + scn + " : " + time + ", " + totalBytes + " bytes.", channel);
62    }
63  }
